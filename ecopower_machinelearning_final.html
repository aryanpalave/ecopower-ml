In [30]:
from sklearn.model_selection import train_test_split
from sklearn.preprocessing import StandardScaler, OneHotEncoder
from sklearn.compose import ColumnTransformer
from sklearn.pipeline import Pipeline

df['HighEmissionFlag'] = (df['Emissions_CO2'] > 75).astype(int)
X = df.drop(columns=['Emissions_CO2', 'HighEmissionFlag'])
y = df['Emissions_CO2']
y_class = df['HighEmissionFlag']

categorical = ['Aircraft', 'Airport', 'InfraType']
numerical = ['Temperature', 'Humidity', 'WindSpeed', 'ScheduleTime', 'UsedAPUMin', 'EfficiencyRatio', 'TempHumidityIndex']

preprocessor = ColumnTransformer([
    ('num', StandardScaler(), numerical),
    ('cat', OneHotEncoder(handle_unknown='ignore'), categorical)
])

X_train, X_test, y_train, y_test = train_test_split(X, y, test_size=0.2, random_state=7)
X_train_cls, X_test_cls, y_train_cls, y_test_cls = train_test_split(X, y_class, test_size=0.2, random_state=7)

from IPython.display import display, Latex
from sklearn.metrics import mean_absolute_error, mean_squared_error
from sklearn.pipeline import Pipeline
import numpy as np
from xgboost import XGBRegressor
from lightgbm import LGBMRegressor



# Define models
models = {
    "Random Forest": RandomForestRegressor(n_estimators=100),
    "Gradient Boosting": GradientBoostingRegressor(),
    "Ridge Regression": Ridge(alpha=1.0),
    "XGBoost": XGBRegressor(),
    "LightGBM": LGBMRegressor()
}

# Store results
results = []

for name, model_obj in models.items():
    pipe = Pipeline([('preprocessor', preprocessor), ('regressor', model_obj)])
    pipe.fit(X_train, y_train)
    preds = pipe.predict(X_test)
    mae = mean_absolute_error(y_test, preds)
    rmse = np.sqrt(mean_squared_error(y_test, preds))
    results.append((name, mae, rmse))

import gradio as gr
import pandas as pd

# Custom theme with green primary color
theme = gr.themes.Base(
    primary_hue="green",
    neutral_hue="gray"
)

def predict_emission(aircraft, airport, temperature, humidity, windspeed, schedule_time, infra_type, apu_minutes):
    df_input = pd.DataFrame([{
        'Aircraft': aircraft,
        'Airport': airport,
        'Temperature': temperature,
        'Humidity': humidity,
        'WindSpeed': windspeed,
        'ScheduleTime': schedule_time,
        'InfraType': infra_type,
        'UsedAPUMin': apu_minutes,
        'EfficiencyRatio': apu_minutes / (apu_minutes + 1e-5),
        'TempHumidityIndex': temperature * humidity
    }])

    emission = loaded_model.predict(df_input)[0]

    miles_driven = emission / 0.404
    gallons_jetfuel = emission / 9.57
    trees_year = emission / 21

    return (
        f"Predicted Emissions: {emission:.2f} kg CO₂\n\n"
        f"🛣️ ~ {miles_driven:.0f} miles driven (gasoline car)\n"
        f"✈️ ~ {gallons_jetfuel:.1f} gallons of jet fuel burned\n"
        f"🌳 ~ Carbon absorbed by {trees_year:.1f} trees in 1 year"
    )

with gr.Blocks(theme=theme, css=".gradio-container { max-width: 100% !important; }") as demo:
    gr.Markdown("# EcoPower Emission Predictor")
    with gr.Row():
        with gr.Column(scale=1):
            temp = gr.Slider(0, 40, value=20, label="Temperature (°C)")
            hum = gr.Slider(10, 100, value=60, label="Humidity (%)")
            wind = gr.Slider(0, 30, value=10, label="Wind Speed (km/h)")
            time = gr.Slider(30, 180, value=90, label="Turnaround Time (min)")
            apu = gr.Slider(0, 90, value=40, label="APU Minutes Used")

        with gr.Column(scale=1):
            aircraft = gr.Dropdown(['A320', 'B737', 'A350', 'B787'], label="Aircraft Type")
            airport = gr.Dropdown(['JFK', 'LAX', 'SEA', 'ATL'], label="Airport")
            infra = gr.Dropdown(['Basic', 'Moderate', 'Advanced'], label="Infrastructure Type")
            submit = gr.Button("Submit")
            clear = gr.Button("Clear")

        with gr.Column(scale=1):
            output = gr.Textbox(label="Environmental Impact", lines=6)

    submit.click(
        predict_emission,
        inputs=[aircraft, airport, temp, hum, wind, time, infra, apu],
        outputs=output
    )
    
    clear.click(
        lambda: "",
        inputs=[],
        outputs=output
    )

This interactive emissions predictor provides a concrete foundation for the EcoPower concept by showcasing how machine learning can intelligently estimate environmental impact based on real aircraft turnaround parameters. By translating raw APU data into meaningful sustainability metrics—such as equivalent miles driven or trees required to offset emissions—it demonstrates the technical insights necessary to optimize the APU. Earlier in the pipeline, various machine learning models—including Random Forest, Gradient Boosting, Ridge Regression, XGBoost, and LightGBM—were trained and evaluated to identify the most accurate and robust estimator for predicting CO₂ emissions. These models not only demonstrated high performance (with the best achieving low MAE and RMSE values), but also provided explainable outputs that informed feature importance and model behavior. Together, the trained ML models and interactive interface form a complete loop: data-driven prediction, real-time visualization, and actionable insight.
